Class RTM::RTMAuth

  1. lib/rtm/auth.rb
Parent: RTM::RTMBase

Implements authorization related tasks. These are to be used one-time only when the user first uses your application. 

auth = RTMAuth.new(api_key,secret)
url = auth.get_auth_url
# send user to url
# When they have done so
token = auth.get_token
# object auth no longer needed; use token with RTM

    def get_frob
      response = http.get(url_for('rtm.auth.getFrob'))
      verify(response)
      @frob = response['rsp']['frob']
      @frob
    end

    def get_token
      response = http.get(url_for('rtm.auth.getToken', { 'frob' => @frob }))
      verify(response)
      response['rsp']['auth']['token']
    end
